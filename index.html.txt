                <button onclick="handleConversion('easy')" id="btn-easy" class="flex items-center justify-center gap-2 bg-blue-600 text-white px-6 py-4 rounded-2xl font-bold hover:bg-blue-700 transition-all shadow-lg active:scale-95">

                    <i data-lucide="languages" class="w-5 h-5"></i> ミャンマー語
                </button>
                <button onclick="handleConversion('nepali')" id="btn-nepali" class="flex items-center justify-center gap-2 bg-orange-500 text-white px-6 py-4 rounded-2xl font-bold hover:bg-orange-600 transition-all shadow-lg active:scale-95">
                    <i data-lucide="languages" class="w-5 h-5"></i> ネパール語
                </button>

    <script>
        // --- 設定 ---
        const API_KEY = "AIzaSyCRIgIcLoheQKzeSY-4sGWyZxhW37NklqE"; 
        const MODEL_NAME = "gemini-1.5-flash"; // 安定版の1.5 Flashを推奨


        let currentMode = '';


        async function handleConversion(mode) {
            const input = document.getElementById('inputText').value.trim();
            if (!input || !API_KEY) return;


            currentMode = mode;
            showLoading(true);


            const systemPrompts = {
                easy: `あなたは「やさしい日本語」の専門家です。指示：入力を「やさしい日本語」に変換し、全ての漢字にHTMLの<ruby>タグでルビを振ってください。ルール：ハサミの法則（はっきり・さいごまで・みじかく）を厳守。敬語を避け「です・ます・ください」に統一。出力は必ず以下のJSON形式のみ。{"converted_text": "文章（<ruby>漢<rt>かん</rt>字<rt>じ</rt></ruby>付き）"}`,
                myanmar: `翻訳：日本語からミャンマー語へ。JSON：{"converted_text": "翻訳結果のみ"}`,
                nepali: `翻訳：日本語からネパール語へ。JSON：{"converted_text": "翻訳結果のみ"}`
            };


            try {
                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/${MODEL_NAME}:generateContent?key=${API_KEY}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{ parts: [{ text: input }] }],
                        systemInstruction: { parts: [{ text: systemPrompts[mode] }] },
                        generationConfig: { responseMimeType: "application/json", temperature: 0 }
                    })
                });


                const data = await response.json();
                const result = JSON.parse(data.candidates[0].content.parts[0].text);
                
                document.getElementById('resultCard').classList.remove('hidden');
                document.getElementById('outputText').innerHTML = result.converted_text;
                showLoading(false);
            } catch (error) {
                console.error(error);
                alert("エラーが発生しました。APIキーが正しいか確認してください。");
                showLoading(false);
            }
        }


        function showLoading(isLoading) {
            const btn = document.getElementById(`btn-${currentMode}`);
            if (isLoading) {
                document.getElementById('outputText').innerHTML = '<p class="text-blue-600 animate-pulse text-center">翻訳中...</p>';
                document.getElementById('resultCard').classList.remove('hidden');
            }
        }


        function copyToClipboard() {
            let text = document.getElementById('outputText').innerHTML;
            // Rubyタグを「漢字（かんじ）」形式に変換
            text = text.replace(/<ruby>(.*?)<rt>(.*?)<\/rt><\/ruby>/g, '$1（$2）').replace(/<[^>]*>?/gm, '');
            
            navigator.clipboard.writeText(text).then(() => {
                const toast = document.getElementById('toast');
                toast.innerText = "コピーしました！";
                toast.classList.remove('hidden');
                setTimeout(() => toast.classList.add('hidden'), 3000);
            });
        }


        function clearInput() {
            if(confirm("入力を消去しますか？")) {
                document.getElementById('inputText').value = "";
                document.getElementById('resultCard').classList.add('hidden');
            }
        }


        // アイコンの初期化
        lucide.createIcons();
    </script>
